```cpp
#include <iostream>

class Entity {
public:
    float X, Y;

    Entity() {
        X = 0.0f;
        Y = 0.0f;
    }

    Entity(float x, float y) {
        X = x;
        Y = y;
    }

    void Print() {
        std::cout << X << ", " << Y << std::endl; 
    }
}

int main() {
    Entity e(10.0f, 5.0f);
    std::cout << e.X << std::endl;
    e.Print();
    std::cin.get();
}
```

#### How to make constructor private so you can't instantiate class
```cpp
#include <iostream>

class Log {

// one of the methods to disable constructor
private:
    Log() {}

    // other method to disable constructor:
    Log() = delete();

public:
    static void Write() {

    }
}

int main() {
    Log::Write();
    Log l; // will produce error

    std::cin.get();
}
```
